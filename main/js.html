دستور find بررسی می کند و زمانی که به اولین مورد رسید که شرط درست بود آن را برمی گرداند و حلقه پیمایش خودش را متوقف می کند. (فقط یک مورد را برمی گرداند)

    {name:"ipad" , category:"mobile"},
    {name:"LG" , category:"TV"},
    {name:"iphone" , category:"Phone"},
    {name:"Super Vision" , category:"TV"},
];

 

 


const result = products.filter(function(product){
    return product.category == "TV";
});

                                         //1: {name: "Super Vision", category: "TV"}

------- alias - default ------

گاهی اسم هایی که اکسپورت یا اینپورت کرده این در فایلی که داریم اینپورت می کنیم وجود دارد و یا به هر دلیلی ممکن است با ارور مواجه شویم که در این صورت از alias برای تعیین یک نام مستعار استفاده می کنیم تا مشکل برطرف شود.

const sum = 20;

 

export {sum as sum1}

 

 

in main file

 

import {sum1} from "./module.js"

دقیقا برای اینپورت هم می توانیم همین کار را انجام دهیم

 

 

import {sum as sum1} from "./module.js"

گاهی تعداد متغییرهایی که اکسپورت شده است زیاد است و ما می خواهیم همه آن ها را دریافت کنیم که در این صورت از روش زیر استفاده می کنیم.

import *  as  app   from    "./module.js"

 

console.log(app.array);

function activateFullscreen(element) {
          if(element.requestFullscreen) {
            element.requestFullscreen();        // W3C spec
          }
          else if (element.mozRequestFullScreen) {
            element.mozRequestFullScreen();     // Firefox
          }
          else if (element.webkitRequestFullscreen) {
            element.webkitRequestFullscreen();        // Safari
          }
          else if(element.msRequestFullscreen) {
            element.msRequestFullscreen();               // IE/Edge
          }
}

 

function deactivateFullscreen() {
          if(document.exitFullscreen) {
            document.exitFullscreen();
          } else if (document.mozCancelFullScreen) {
            document.mozCancelFullScreen();
          } else if (document.webkitExitFullscreen) {
            document.webkitExitFullscreen();
          }else if (document.msExitFullscreen) {
            document.msExitFullscreen();
          }
}

function toggle(element){
            if (document.fullscreenElement || document.mozFullScreenElement || document.webkitFullscreenElement || document.msFullscreenElement) {
                deactivateFullscreen();
            } else {
                activateFullscreen(element);
            }
 }

window.number_format = function (number, decimals, dec_point, thousands_point) {

        if (number == null || !isFinite(number)) {
            throw new TypeError("number is not valid");
        }

        if (!decimals) {
            var len = number.toString().split('.').length;
            decimals = len > 1 ? len : 0;
        }

        if (!dec_point) {
            dec_point = '.';
        }

        if (!thousands_point) {
            thousands_point = ',';
        }

        number = parseFloat(number).toFixed(decimals);

        number = number.replace(".", dec_point);

        var splitNum = number.split(dec_point);
        splitNum[0] = splitNum[0].replace(/\B(?=(\d{3})+(?!\d))/g, thousands_point);
        number = splitNum.join(dec_point);

        return number;
    }

 

// for use

console.log(number_format (300000));     //  "300,000"

console.log(number_format (300000 , 2));     //  "300,000.00"

console.log(number_format (300000, 2 , "." , "/"));     // "300/000.00"
